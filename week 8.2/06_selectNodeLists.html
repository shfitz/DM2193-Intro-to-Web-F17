Selecting items from NodeLists

item() syntax

var elements = document.getElementsByClassName("hotness");
        if(elements.length >= 1){
            var firstElem = elements.item(0);
        }
var elements = document.getElementsByClassName("hotness");
        if(elements.length >= 1){
            var allItems = [];
            for(var i = 0; i<elements.length; i++){
                allItems[i]=elements.item(i);
        }
}
